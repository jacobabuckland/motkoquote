SQL — copy into the Supabase SQL editor

1 · Daily signups

select date_trunc('day', created_at) as day,
       count(*) as signups
from events
where event_name = 'signed_up'
group by 1
order by 1 desc;

2 · Signup → quote_sent → payment_received funnel

with signed as (
  select distinct user_id from events
  where event_name = 'signed_up' and user_id is not null
),
sent as (
  select distinct user_id from events
  where event_name = 'quote_sent' and user_id is not null
),
paid as (
  -- payment_received is anonymous (webhook); join via shared quote_id
  select distinct s.user_id
  from events p
  join events s
    on s.event_name = 'quote_sent'
   and s.properties->>'quote_id' = p.properties->>'quote_id'
  where p.event_name = 'payment_received'
)
select
  (select count(*) from signed) as signed_up,
  (select count(*) from sent where user_id in (select user_id from signed)) as reached_quote_sent,
  (select count(*) from paid where user_id in (select user_id from signed)) as reached_payment;

3 · All events for a given user

select event_name, properties, created_at
from events
where user_id = '00000000-0000-0000-0000-000000000000'
order by created_at desc;
